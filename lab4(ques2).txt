2.WAP to compute the BMI Index of the person and print the BMI values as per the following ranges. You can use the following formula to compute BMI= weight(kgs)/Height(Mts)*Height(Mts).

	        BMI
Starvation	<15
Anorexic	15.1 to 17.5
Underweight	17.6 to 18.5
Ideal	        18.6 to 24.9
Overweight	25 to 25.9
Obese	        30 to 39.9
Morbidity Obese	40.0 above


#include<stdio.h>
#include<conio.h>
int main()
{
    float weight,height,BMI;
    printf("enter weight in kg:");
    scanf("%f",&weight);
    printf("enter height in meters:");
    scanf("%f",&height);
    BMI=weight/(height*height);
    printf("BMI is %f\n",BMI);
    switch((BMI<=15)?1:(BMI>=15.1 && BMI<=17.5)?2:(BMI>=17.6 && BMI<=18.5)?3:(BMI>=18.6 && BMI<=24.9)?4:(BMI>=25 && BMI<=29.9)?6:7)
    {
        case 1:
        printf("suffering from starvation\n");
        break;
        case 2:
        printf("suffering from anorexic\n");
        break;
        case 3:
        printf("underweight\n");
        break;
        case 4:
        printf("ideal weight\n");
        break;
        case 5:
        printf("overweight\n");
        break;
        case 6:
        printf("suffering from obeseity\n");
        break;
        case 7:
        printf("suffering from morbid obeseity\n");
        break;
        default:
        printf("invalid BMI\n");
    }
    return 0;
}
